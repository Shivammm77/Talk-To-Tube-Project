const API_URL = "http://localhost:8000";

const apiService = {
  async getVideoInfo(url) {
    try {
      const res = await fetch(`${API_URL}/video_info?video_url=${encodeURIComponent(url)}`);
      if (!res.ok) throw new Error("Failed to fetch video info");
      return await res.json();
    } catch (error) {
      console.warn("Backend unreachable, using simulation:", error);
      await new Promise(r => setTimeout(r, 1500));
      return {
        title: "Build a Modern AI App with React & FastAPI",
        thumbnail: "https://i.ytimg.com/vi/7sB052Pz0sQ/maxresdefault.jpg",
        channel: "TechMastery",
        views: "1.2M"
      };
    }
  },

  async askQuestion(url, queryText) {
    try {
      const res = await fetch(`${API_URL}/ask/?url=${encodeURIComponent(url)}`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ query: queryText })
      });
      if (!res.ok) throw new Error("Failed to get answer");
      return await res.json();
    } catch (error) {
      console.warn("Backend unreachable, using simulation:", error);
      await new Promise(r => setTimeout(r, 2000));
      return { 
        answer: `This is a simulated response for: "${queryText}". \n\nThe video explains that FastAPI automatically generates OpenAPI documentation and uses Pydantic for data validation, making it significantly faster than Flask for this use case.` 
      };
    }
  }
};

// 2. Auth Modal
const AuthModal = ({ isOpen, onClose }) => {

  const [error, setError] = useState('');
  const [loading, setLoading] = useState(false);

  if (!isOpen) return null;

  const handleAuth = async (e) => {
    e.preventDefault();
    setError('');
    setLoading(true);
    try {
      if (isLogin) {
        await signInWithEmailAndPassword(auth, email, password);
      } else {
        const userCred = await createUserWithEmailAndPassword(auth, email, password);
        await updateProfile(userCred.user, {
           photoURL: "https://api.dicebear.com/7.x/avataaars/svg?seed=" + userCred.user.uid
        });

      setError(err.message.replace('Firebase: ', ''));

      setLoading(false);
    }

    <div className="fixed inset-0 z-50 flex items-center justify-center p-4 bg-black/90 backdrop-blur-md animate-in fade-in duration-300">

          </button>

          <div className="text-center mb-8 mt-4">
            <h2 className="text-3xl font-bold text-white mb-2 tracking-tight">
              {isLogin ? 'Welcome Back' : 'Join TalkToTube'}

// 4. Main Chat Interface
const ChatApp = ({ user }) => {
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState('');
  const [videoUrl, setVideoUrl] = useState('');
  const [videoData, setVideoData] = useState(null);
  const [loading, setLoading] = useState(false);
  const [history, setHistory] = useState([]);
  const [showProfile, setShowProfile] = useState(false);
  const [isSidebarOpen, setIsSidebarOpen] = useState(true);
  
  const messagesEndRef = useRef(null);
  const inputRef = useRef(null);

  useEffect(() => {
    if (!user) return;
    const q = query(
      collection(db, 'artifacts', appId, 'users', user.uid, 'chat_history'), 
      orderBy('createdAt', 'desc'),
      limit(20)
    );
    const unsubscribe = onSnapshot(q, (snapshot) => {
      setHistory(snapshot.docs.map(doc => ({ id: doc.id, ...doc.data() })));
    });
    return () => unsubscribe();
  }, [user]);

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages, loading]);

  const handleVideoLoad = async () => {
    if (!videoUrl) return;
    setLoading(true);
    try {
      const data = await apiService.getVideoInfo(videoUrl);
      setVideoData(data);
      
      // Save to History immediately upon loading
      await addDoc(collection(db, 'artifacts', appId, 'users', user.uid, 'chat_history'), {
        title: data.title,
        thumbnail: data.thumbnail,
        url: videoUrl,
        createdAt: serverTimestamp()
      });

      // Set initial greeting
      setMessages([{ 
        id: Date.now(), 
        type: 'bot', 
        text: `I've analyzed **${data.title}**. \n\nWhat would you like to know about this video?` 
      }]);
      
    } catch (err) {
      setMessages([{ id: Date.now(), type: 'error', text: err.message }]);
    } finally {
      setLoading(false);
    }
  };

  const handleSendMessage = async () => {
    if (!input.trim() || !videoUrl) return;

    const userMsg = { id: Date.now(), type: 'user', text: input };
    setMessages(prev => [...prev, userMsg]);
    setInput('');
    setLoading(true);

    try {
      const response = await apiService.askQuestion(videoUrl, input);
      const botMsg = { 
        id: Date.now() + 1, 
        type: 'bot', 
        text: response.answer || JSON.stringify(response) 
      };
      setMessages(prev => [...prev, botMsg]);
    } catch (err) {
      setMessages(prev => [...prev, { id: Date.now(), type: 'error', text: "Failed to get answer. Is the backend running?" }]);
    } finally {
      setLoading(false);
    }
  };

  const loadHistoryItem = (item) => {
    setVideoUrl(item.url);
    setVideoData({
      title: item.title,
      thumbnail: item.thumbnail,
      channel: "Restored History"
    });
    setMessages([{ 
      id: Date.now(), 
      type: 'bot', 
      text: `Welcome back to **${item.title}**. Continue where you left off.` 
    }]);
  };

  const resetChat = () => {
    setVideoData(null);
    setVideoUrl('');
    setMessages([]);
    setInput('');
  };

  return (
    <div className="flex h-screen bg-black text-white overflow-hidden font-sans">
      
      {/* MAIN AREA */}
      <div className="flex-1 flex flex-col h-full relative z-0 transition-all duration-300">
        
        {/* Top Navbar */}
        <header className="h-16 border-b border-zinc-900 flex items-center justify-between px-6 bg-black/80 backdrop-blur-xl sticky top-0 z-30">
          <div className="flex items-center gap-4">
             <button onClick={resetChat} className="hover:opacity-80 transition-opacity">
                <Logo />
             </button>
             {videoData && (
                <div className="hidden md:flex items-center gap-2 px-3 py-1 rounded-full bg-zinc-900 border border-zinc-800">
                   <Youtube className="w-4 h-4 text-red-500" />
                   <span className="text-xs text-zinc-300 max-w-[200px] truncate">{videoData.title}</span>
                </div>
             )}
          </div>
          <div className="flex items-center gap-4">
             {videoData && (
               <button 
                 onClick={resetChat}
                 className="hidden md:flex items-center gap-2 px-3 py-1.5 text-xs font-medium text-zinc-400 hover:text-white hover:bg-zinc-900 rounded-lg transition-colors"
               >
                 <Plus className="w-4 h-4" /> New Chat
               </button>
             )}
             <button onClick={() => setShowProfile(true)} className="group relative">
               <div className="w-9 h-9 rounded-full bg-gradient-to-br from-red-500 to-orange-500 p-[2px]">
                  <img src={user.photoURL || "https://api.dicebear.com/7.x/avataaars/svg?seed=" + user.uid} alt="Profile" className="w-full h-full rounded-full bg-black object-cover" />
               </div>
             </button>
             <button onClick={() => setIsSidebarOpen(!isSidebarOpen)} className="p-2 text-zinc-400 hover:text-white md:hidden">
               <History className="w-5 h-5" />
             </button>
          </div>
        </header>

        {/* --- VIDEO CONTEXT HEADER (Visible only when video loads) --- */}
        {videoData && (
          <div className="flex-shrink-0 bg-zinc-950/50 border-b border-zinc-900 p-4 animate-in slide-in-from-top-4 duration-500">
            <div className="max-w-3xl mx-auto flex gap-4 items-center">
               <div className="relative w-32 md:w-40 aspect-video rounded-xl overflow-hidden shadow-lg ring-1 ring-zinc-800 group">
                  <img src={videoData.thumbnail} alt="Active Video" className="w-full h-full object-cover group-hover:scale-110 transition-transform duration-700" />
                  <div className="absolute inset-0 bg-black/10 group-hover:bg-transparent" />
                  <div className="absolute inset-0 flex items-center justify-center opacity-0 group-hover:opacity-100 transition-opacity">
                     <a href={videoUrl} target="_blank" rel="noreferrer" className="bg-red-600/90 p-2 rounded-full text-white shadow-lg backdrop-blur-sm hover:scale-110 transition-transform">
                        <Play className="w-4 h-4 fill-white" />
                     </a>
                  </div>
               </div>
               <div className="flex-1 min-w-0">
                  <h2 className="text-lg md:text-xl font-bold text-white truncate leading-tight">{videoData.title}</h2>
                  <div className="flex items-center gap-3 mt-2 text-sm text-zinc-500">
                     <span className="flex items-center gap-1.5"><User className="w-3.5 h-3.5" /> {videoData.channel || "Unknown Channel"}</span>

                  </div>
               </div>
            </div>
          </div>
        )}

        {/* Chat Area */}
        <div className="flex-1 overflow-y-auto p-4 md:p-6 scrollbar-thin scrollbar-thumb-zinc-800 scrollbar-track-transparent bg-gradient-to-b from-black to-zinc-950">
          <div className="max-w-3xl mx-auto space-y-6">
            
            {/* EMPTY STATE - INITIAL URL INPUT */}
            {!videoData && (
              <div className="flex flex-col items-center justify-center min-h-[70vh] text-center animate-in fade-in zoom-in duration-500">
                 <div className="mb-8 relative group">
                    <div className="absolute inset-0 bg-red-600 blur-[40px] opacity-20 group-hover:opacity-30 transition-opacity rounded-full" />
                    <div className="relative w-24 h-24 bg-zinc-900 rounded-3xl flex items-center justify-center border border-zinc-800 shadow-2xl shadow-red-900/10">
                       <Youtube className="w-12 h-12 text-white group-hover:text-red-500 transition-colors duration-300" />
                    </div>
                 </div>
                 
                 <h1 className="text-4xl font-bold mb-3 tracking-tight bg-clip-text text-transparent bg-gradient-to-b from-white to-zinc-500">
                   What are we watching?
                 </h1>
                 <p className="text-zinc-500 text-lg mb-10 max-w-md">
                   Paste a YouTube URL to unlock the power of AI analysis and chat with the video.
                 </p>
                 
                 <div className="w-full max-w-xl relative group z-10">
                    <div className="absolute -inset-0.5 bg-gradient-to-r from-red-500 to-orange-500 rounded-2xl blur opacity-30 group-hover:opacity-50 transition-opacity duration-500" />
                    <div className="relative flex items-center bg-zinc-950 border border-zinc-800 rounded-2xl p-2 pl-5 shadow-2xl">
                      <input 
                        value={videoUrl}
                        onChange={(e) => setVideoUrl(e.target.value)}
                        placeholder="https://youtube.com/watch?v=..."
                        className="flex-1 bg-transparent border-none focus:outline-none text-white placeholder:text-zinc-600 text-lg h-12"
                      />
                      <button 
                        onClick={handleVideoLoad}
                        disabled={!videoUrl || loading}
                        className="h-12 px-8 bg-white text-black font-bold rounded-xl hover:bg-zinc-200 transition-all hover:scale-105 active:scale-95 disabled:opacity-50 disabled:hover:scale-100 flex items-center gap-2"
                      >
                        {loading ? <Loader2 className="w-5 h-5 animate-spin" /> : <>Load <ArrowRight className="w-4 h-4" /></>}
                      </button>
                    </div>
                 </div>
              </div>
            )}

            {/* MESSAGES */}
            {messages.map((msg) => (
              <div key={msg.id} className={`flex ${msg.type === 'user' ? 'justify-end' : 'justify-start'} animate-in slide-in-from-bottom-4 duration-300`}>
                <div className={`max-w-[85%] md:max-w-[75%] rounded-2xl p-5 shadow-sm ${
                  msg.type === 'user' 
                    ? 'bg-zinc-800 text-white rounded-tr-sm' 
                    : msg.type === 'error' 
                      ? 'bg-red-950/30 text-red-200 border border-red-900/50' 
                      : 'bg-zinc-900/50 border border-zinc-800/50 text-zinc-200'
                }`}>
                  {msg.type === 'bot' && (
                    <div className="flex items-center gap-2 mb-2">
                       <Sparkles className="w-4 h-4 text-orange-500" />
                       <span className="text-xs font-bold text-orange-500 uppercase tracking-wider">AI Insight</span>
                    </div>
                  )}
                  <p className="whitespace-pre-wrap leading-relaxed text-[15px]">{msg.text}</p>
                </div>
              </div>
            ))}
            
            {loading && messages.length > 0 && (
               <div className="flex justify-start animate-in fade-in">
                 <div className="bg-zinc-900/50 border border-zinc-800 rounded-2xl px-5 py-4 flex items-center gap-3">
                   <div className="flex gap-1">
                      <span className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce [animation-delay:-0.3s]" />
                      <span className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce [animation-delay:-0.15s]" />
                      <span className="w-2 h-2 bg-zinc-500 rounded-full animate-bounce" />
                   </div>
                   <span className="text-sm text-zinc-500">Analyzing...</span>
                 </div>
               </div>
            )}
            
            <div ref={messagesEndRef} className="h-4" />
          </div>
        </div>

        {/* Input Bar (Only visible if video is loaded) */}
        {videoData && (
          <div className="p-4 md:p-6 max-w-3xl mx-auto w-full bg-gradient-to-t from-black to-transparent">
            <div className="relative flex items-center bg-zinc-900 border border-zinc-700/50 rounded-full p-2 pl-6 shadow-2xl shadow-black/50 focus-within:ring-2 focus-within:ring-zinc-700 focus-within:bg-zinc-900 transition-all">
              <input 
                ref={inputRef}
                value={input}
                onChange={(e) => setInput(e.target.value)}
                onKeyDown={(e) => e.key === 'Enter' && handleSendMessage()}
                placeholder="Ask follow-up questions..."
                className="flex-1 bg-transparent border-none focus:outline-none text-white placeholder:text-zinc-500 h-10"
                autoFocus
              />
              <button 
                onClick={handleSendMessage}
                disabled={!input.trim() || loading}
                className="w-10 h-10 rounded-full bg-white text-black flex items-center justify-center hover:bg-zinc-200 transition-transform active:scale-95 disabled:opacity-50 disabled:hover:bg-white ml-2"
              >
                {loading ? <Loader2 className="w-4 h-4 animate-spin" /> : <Send className="w-4 h-4" />}
              </button>
            </div>
            <p className="text-center mt-3 text-[10px] text-zinc-600">AI can make mistakes. Verify important info.</p>
          </div>
        )}
      </div>

      {/* RIGHT SIDEBAR (HISTORY) */}
      <div className={`${isSidebarOpen ? 'translate-x-0' : 'translate-x-full'} md:translate-x-0 fixed md:relative right-0 top-0 bottom-0 w-80 bg-zinc-950 border-l border-zinc-900 transform transition-transform duration-300 z-40 flex flex-col shadow-2xl md:shadow-none`}>
        <div className="p-5 border-b border-zinc-900 flex items-center justify-between bg-zinc-950">
          <div className="flex items-center gap-2 text-zinc-400">
            <Clock className="w-4 h-4" />
            <h2 className="font-semibold text-xs uppercase tracking-wider">History</h2>
          </div>
          <button onClick={() => setIsSidebarOpen(false)} className="md:hidden text-zinc-500 hover:text-white">
            <X className="w-5 h-5" />
          </button>
        </div>
        
        <div className="flex-1 overflow-y-auto p-3 space-y-3">
          {history.length === 0 ? (
            <div className="flex flex-col items-center justify-center h-full text-zinc-700 space-y-2 opacity-50">
              <History className="w-8 h-8" />
              <p className="text-xs">No chat history found</p>
            </div>
          ) : (
            history.map((item) => (
              <button 
                key={item.id}
                onClick={() => {
                   loadHistoryItem(item);
                   if(window.innerWidth < 768) setIsSidebarOpen(false);
                }}
                className="w-full text-left p-3 rounded-xl bg-zinc-900/40 hover:bg-zinc-900 transition-all border border-transparent hover:border-zinc-800 group relative overflow-hidden"
              >
                <div className="flex gap-3">
                  <div className="w-16 h-12 rounded-lg bg-zinc-800 flex-shrink-0 overflow-hidden relative">
                     {item.thumbnail ? (
                       <img src={item.thumbnail} className="w-full h-full object-cover group-hover:scale-110 transition-transform duration-500" alt="" />
                     ) : (
                       <Youtube className="w-5 h-5 text-zinc-600 m-auto mt-3.5" />
                     )}
                     <div className="absolute inset-0 bg-black/0 group-hover:bg-black/10 transition-colors" />
                  </div>
                  <div className="flex-1 min-w-0 flex flex-col justify-center">
                    <p className="text-sm font-medium text-zinc-300 truncate group-hover:text-white transition-colors">{item.title || "Untitled Chat"}</p>
                    <p className="text-[10px] text-zinc-600 mt-0.5 truncate flex items-center gap-1">
                      {item.createdAt?.seconds ? new Date(item.createdAt.seconds * 1000).toLocaleDateString() : 'Just now'}
                    </p>
                  </div>
                </div>
              </button>
            ))
          )}
        </div>
      </div>

      <ProfileModal 
        user={user} 
        isOpen={showProfile} 
        onClose={() => setShowProfile(false)} 
      />
      
      {/* CSS for Scrolling Animation */}
      <style>{`
        @keyframes scroll-left {
          0% { transform: translateX(0); }
          100% { transform: translateX(-50%); }
        }
        .animate-scroll-left {
          animation: scroll-left 40s linear infinite;
        }
        .hover\\:pause-animation:hover {
          animation-play-state: paused;
        }
        .mask-linear-fade {
          mask-image: linear-gradient(to right, transparent, black 10%, black 90%, transparent);
        }
      `}</style>
    </div>
  );
};

// 5. App Container
export default function App() {
  const [user, setUser] = useState(null);
  const [loading, setLoading] = useState(true);
  const [view, setView] = useState('landing');
  const [showAuthModal, setShowAuthModal] = useState(false);

  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, (currentUser) => {
      setUser(currentUser);
      if (currentUser) {
        setView('chat');
      } else {
        if (view === 'chat') setView('landing');
      }
      setLoading(false);
    });
    return () => unsubscribe();
  }, [view]);

  if (loading) {
    return (
      <div className="h-screen w-screen bg-black flex items-center justify-center">
        <div className="flex flex-col items-center gap-6">
          <div className="relative">
             <div className="w-12 h-12 rounded-full border-4 border-zinc-800 border-t-red-500 animate-spin" />
             <div className="absolute inset-0 flex items-center justify-center">
                <div className="w-2 h-2 bg-white rounded-full" />
             </div>
          </div>
          <span className="text-zinc-500 font-medium tracking-widest text-xs uppercase animate-pulse">Initializing</span>
        </div>
      </div>
    );
  }

  return (
    <>
      <style>{`
        @keyframes fade-in-up {
          from { opacity: 0; transform: translateY(20px); }
          to { opacity: 1; transform: translateY(0); }
        }
        .animate-fade-in-up {
          animation: fade-in-up 0.8s cubic-bezier(0.16, 1, 0.3, 1) forwards;
        }
      `}</style>
      
      {view === 'landing' && (
        <LandingPage 
          onGetStarted={() => setShowAuthModal(true)} 
        />
      )}

      {view === 'chat' && user && (
        <ChatApp user={user} />
      )}

      <AuthModal 
        isOpen={showAuthModal} 
        onClose={() => setShowAuthModal(false)} 
      />
    </>
  );
}
